    private void jToggleButton5ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jToggleButton5ActionPerformed
        // TODO add your handling code here:
        int total = Integer.parseInt(tfTotal.getText());
        int money = Integer.parseInt(tfYourMoney.getText());
        if(money == 0) {
            JOptionPane.showMessageDialog(null,"Masukan Uang Anda Terlebih Dahulu");
        }
        else if (total > money) {
            JOptionPane.showMessageDialog(null, "Uang Anda Kurang");          
        } else {
            checkout checkoutObj = new checkout();
            checkoutObj.setTotal(Integer.parseInt(tfTotal.getText()));
            checkoutObj.setMoney(Integer.parseInt(tfYourMoney.getText()));
            checkoutObj.checkoutMethod(checkoutObj.getMoney(), checkoutObj.getTotal());
            tfMoneyBack.setText(Integer.toString(checkoutObj.getMoneyBack()));
            JOptionPane.showMessageDialog(null, "Pembayaran Berhasil, Uang Kembalian Anda Adalah Rp." + " " + checkoutObj.getMoneyBack() + " " + "\n Terima Kasih Sudah Datang");
            home newHome = new home();
            newHome.setVisible(true);
            this.dispose();
        }
    }//GEN-LAST:event_jToggleButton5ActionPerformed
